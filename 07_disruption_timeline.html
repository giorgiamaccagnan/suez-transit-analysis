Chart 7 — Timeline of Convergent Disruptions to Global Shipping (2020–2026)¶

Thesis Section: 1.4 — The Age of Convergent Disruptions

Data Sources:

  • UNCTAD, "Maritime Transport in Times of Polycrisis," TD/B/C.I/MEM.7/32, August 2024
  • OECD, "Risks and Resilience in Global Trade," 2024
  • Various sources cited in Section 1.4 footnotes

Description: Gantt-style timeline showing overlapping disruption events from 2020 to 2026, illustrating the polycrisis concept — each crisis overlapping with the next, with no recovery period between them.

import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.dates as mdates
import matplotlib.patches as mpatches
from datetime import datetime
import seaborn as sns

sns.set_theme(style="white", font="serif")
plt.rcParams.update({
    'font.family': 'serif',
    'font.serif': ['Times New Roman', 'DejaVu Serif'],
    'font.size': 11,
    'axes.labelsize': 12,
    'axes.titlesize': 14,
    'figure.dpi': 150,
    'savefig.dpi': 300,
    'savefig.bbox': 'tight'
})

events = [
    {'event': 'COVID-19 Pandemic', 'start': '2020-01-01', 'end': '2022-06-30',
     'color': '#c00000', 'detail': 'Port closures · Container shortage · SCFI $14,000+'},
    {'event': 'Ever Given Blockage', 'start': '2021-03-23', 'end': '2021-03-29',
     'color': '#e67e22', 'detail': '6-day Suez block · 400+ vessels · $9.6B/day'},
    {'event': 'Russia-Ukraine War', 'start': '2022-02-24', 'end': '2026-05-31',
     'color': '#7f8c8d', 'detail': 'Energy crisis · Grain exports · Airspace closure'},
    {'event': 'Panama Canal Drought', 'start': '2023-07-01', 'end': '2024-06-30',
     'color': '#2e75b6', 'detail': 'Transits −49% · Draft restrictions'},
    {'event': 'Red Sea / Houthi Crisis', 'start': '2023-11-19', 'end': '2026-05-31',
     'color': '#d4a017', 'detail': 'Suez traffic −70% · Cape rerouting'},
    {'event': 'Strait of Hormuz Escalation', 'start': '2026-03-01', 'end': '2026-05-31',
     'color': '#8e44ad', 'detail': 'Freight $4,131/TEU · Truck bridges'},
]

fig, ax = plt.subplots(figsize=(14, 5.5))

for i, ev in enumerate(reversed(events)):
    start = datetime.strptime(ev['start'], '%Y-%m-%d')
    end = datetime.strptime(ev['end'], '%Y-%m-%d')
    duration = (end - start).days
    
    ax.barh(i, duration, left=mdates.date2num(start), height=0.55, 
            color=ev['color'], alpha=0.85, edgecolor='white', linewidth=1)
    
    # Event name on bar
    mid = mdates.date2num(start) + duration / 2
    ax.text(mid, i, ev['event'], ha='center', va='center', 
            fontsize=9.5, fontweight='bold', color='white',
            bbox=dict(boxstyle='round,pad=0.2', facecolor=ev['color'], alpha=0.7))
    
    # Detail text to the right
    ax.text(mdates.date2num(end) + 15, i, ev['detail'], ha='left', va='center',
            fontsize=8.5, color='#555555', style='italic')

ax.set_yticks([])
ax.xaxis.set_major_formatter(mdates.DateFormatter('%b %Y'))
ax.xaxis.set_major_locator(mdates.MonthLocator(interval=6))
plt.xticks(rotation=45)

ax.set_xlim(mdates.date2num(datetime(2019, 10, 1)), mdates.date2num(datetime(2026, 12, 1)))
ax.set_title('The Age of Convergent Disruptions: Overlapping Crises (2020–2026)', 
            fontweight='bold', pad=15)

# Polycrisis annotation
ax.text(mdates.date2num(datetime(2024, 6, 1)), -0.8, 
        '"The overall impact exceeds the sum of each crisis considered individually" — UNCTAD, 2024',
        fontsize=10, ha='center', style='italic', color='#333333',
        bbox=dict(boxstyle='round,pad=0.5', facecolor='#f5f5f5', edgecolor='#999999', alpha=0.8))

ax.spines['left'].set_visible(False)
ax.spines['right'].set_visible(False)
ax.spines['top'].set_visible(False)

plt.tight_layout()
plt.savefig('../figures/fig_1_4_disruption_timeline.png')
plt.savefig('../figures/fig_1_4_disruption_timeline.pdf')
plt.show()
print("✅ Chart saved to figures/fig_1_4_disruption_timeline.png and .pdf")
