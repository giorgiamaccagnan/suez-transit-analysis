fig = plt.figure(figsize=(15, 8))
gs = GridSpec(1, 2, figure=fig, wspace=0.35)
ax_left  = fig.add_subplot(gs[0])
ax_right = fig.add_subplot(gs[1])


y = np.arange(len(metrics)); bar_h = 0.35
max_v = max(m[2] for m in metrics)
for i, (label, sv, cv, unit) in enumerate(metrics):
    sv_n = sv / max_v * 100; cv_n = cv / max_v * 100
    ax_left.barh(i+bar_h/2, sv_n, height=bar_h, color=BLUE, alpha=0.8)
    ax_left.barh(i-bar_h/2, cv_n, height=bar_h, color=RED,  alpha=0.8)
    u = ' '+unit if unit else ''
    ax_left.text(sv_n+1.5, i+bar_h/2, f'{sv}{u}', va='center', fontsize=8.5, color=BLUE, fontweight='bold')
    ax_left.text(cv_n+1.5, i-bar_h/2, f'{cv}{u}', va='center', fontsize=8.5, color=RED,  fontweight='bold')
    d = f'+{((cv-sv)/sv*100):.0f}%' if cv > sv else f'{((cv-sv)/sv*100):.0f}%'
    dc = RED if cv > sv else GREEN
    ax_left.text(106, i, d, va='center', ha='left', fontsize=9, color=dc, fontweight='bold')

running = 0; bot_l=[]; hgt_l=[]; col_l=[]
for lbl, val, typ in cost_items:
    if   typ=='base':  bot_l.append(0);          hgt_l.append(val);     col_l.append(BLUE);    running = val
    elif typ=='add':   bot_l.append(running);     hgt_l.append(val);     col_l.append('#ef5350'); running += val
    elif typ=='sub':   bot_l.append(running+val); hgt_l.append(abs(val));col_l.append(GREEN);   running += val
    elif typ=='total': bot_l.append(0);           hgt_l.append(running); col_l.append(RED)

total = running
bars_w = ax_right.bar(range(len(cost_items)), hgt_l, bottom=bot_l,
                       color=col_l, alpha=0.85, edgecolor='white', width=0.6)
