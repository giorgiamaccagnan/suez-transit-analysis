# HELPER FUNCTION FOR BARS
# =====================================================

def add_response_bar(x0, x1, y, label, color, height=0.76, alpha=None, fontsize=None):
    width = x1 - x0

    if alpha is None:
        alpha = 0.35 if color == LGRAY else 0.84

    if fontsize is None:
        if width < 1.8:
            fontsize = 5.6
        elif width < 2.8:
            fontsize = 6.2
        elif width < 4.2:
            fontsize = 6.7
        else:
            fontsize = 7.2

    r = mpatches.FancyBboxPatch(
        (x0 + 0.08, y - height / 2),
        width - 0.16,
        height,
        boxstyle='round,pad=0.07',
        facecolor=color,
        edgecolor='white',
        linewidth=0.8,
        alpha=alpha,
        zorder=2
    )

    ax.add_patch(r)

    ax.text(
        (x0 + x1) / 2,
        y,
        label,
        ha='center',
        va='center',
        fontsize=fontsize,
        color='white',
        fontweight='bold',
        zorder=3,
        linespacing=0.9
    )
