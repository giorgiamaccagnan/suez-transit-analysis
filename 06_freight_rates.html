import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.dates as mdates
import matplotlib.patches as mpatches
import seaborn as sns

sns.set_theme(style="whitegrid", font="serif")
plt.rcParams.update({
    'font.family': 'serif',
    'font.serif': ['Times New Roman', 'DejaVu Serif'],
    'font.size': 11,
    'axes.labelsize': 12,
    'axes.titlesize': 14,
    'figure.dpi': 150,
    'savefig.dpi': 300,
    'savefig.bbox': 'tight'
})

df = pd.read_csv('../data/freight_rates.csv')
df['date'] = pd.to_datetime(df['date'])

fig, ax = plt.subplots(figsize=(14, 7))


ax.plot(df['date'], df['scfi_composite'], color='#1f4e79', linewidth=2.5, label='SCFI Composite', zorder=3)
ax.plot(df['date'], df['shanghai_rotterdam'], color='#c00000', linewidth=1.8, linestyle='--', 
        label='Shanghai → Rotterdam', alpha=0.85)
ax.plot(df['date'], df['shanghai_genoa'], color='#e67e22', linewidth=1.8, linestyle='-.', 
        label='Shanghai → Genoa', alpha=0.85)
ax.plot(df['date'], df['shanghai_uswc'], color='#27ae60', linewidth=1.8, linestyle=':', 
        label='Shanghai → US West Coast', alpha=0.85)


ax.fill_between(df['date'], df['scfi_composite'], alpha=0.08, color='#1f4e79')


events = [
    ('COVID-19 Pandemic', '2020-03-01', '2022-06-01', '#c00000', 0.06),
    ('Red Sea Crisis', '2023-11-01', '2026-05-31', '#e67e22', 0.06),
]
for label, start, end, color, alpha in events:
    ax.axvspan(pd.Timestamp(start), pd.Timestamp(end), alpha=alpha, color=color)


events_point = [
    ('Ever Given\nblockage', '2021-03-23', 14800, '#c00000'),
    ('COVID peak\n$14,000/TEU', '2021-09-15', 15500, '#555555'),
    ('Houthi attacks\nbegin', '2023-12-01', 5500, '#e67e22'),
    ('Hormuz escalation\n$4,131/TEU USWC', '2026-03-15', 6500, '#8e44ad'),
]
for label, date, y_text, color in events_point:
    d = pd.Timestamp(date)
    ax.annotate(label, xy=(d, df.loc[df['date'].dt.to_period('M') == d.to_period('M')].iloc[0]['scfi_composite'] if len(df.loc[df['date'].dt.to_period('M') == d.to_period('M')]) > 0 else y_text),
               xytext=(d, y_text),
               fontsize=8.5, ha='center',
               bbox=dict(boxstyle='round,pad=0.3', facecolor='lightyellow', edgecolor=color, alpha=0.85))


ax.axhline(y=830, color='gray', linestyle=':', alpha=0.5)
ax.text(pd.Timestamp('2019-12-15'), 950, 'Pre-COVID baseline ~$830', fontsize=8.5, color='gray')

ax.set_xlabel('Date')
ax.set_ylabel('USD per FEU / TEU', fontweight='bold')
ax.set_title('Container Freight Rates During Major Disruptions (2019–2026)', fontweight='bold', pad=15)

ax.legend(loc='upper left', framealpha=0.9, fontsize=10)
ax.xaxis.set_major_formatter(mdates.DateFormatter('%b %Y'))
ax.xaxis.set_major_locator(mdates.MonthLocator(interval=6))
plt.xticks(rotation=45)
ax.set_ylim(0, 16000)

plt.tight_layout()
plt.savefig('../figures/fig_1_4_freight_rates.png')
plt.savefig('../figures/fig_1_4_freight_rates.pdf')
plt.show()
print("✅ Chart saved to figures/fig_1_4_freight_rates.png and .pdf")
