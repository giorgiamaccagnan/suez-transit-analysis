Description: Trend analysis of reshoring and nearshoring activity from 2015 to 2025, showing acceleration in the post-COVID and Red Sea crisis period. Supports the argument that geopolitical disruptions are driving structural reconfiguration of global supply chains.

fig = plt.figure(figsize=(14, 7.5))
gs  = GridSpec(2, 1, height_ratios=[2, 1], hspace=0.12, figure=fig)
ax1  = fig.add_subplot(gs[0])
ax1b = ax1.twinx()
ax2  = fig.add_subplot(gs[1], sharex=ax1)

years = df['year'].values
bar_c = [GREEN]*len(years)
bars  = ax1.bar(years, df['reshoring_k'], color=bar_c, alpha=0.7,
                width=0.6, edgecolor='white', label='Reshoring jobs (000s)')
ax1.set_ylabel('Reshoring/Nearshoring Jobs (thousands)', fontweight='bold', color=GREEN)
ax1.tick_params(axis='y', labelcolor=GREEN); ax1.set_ylim(0, 550)

ax1b.plot(years, df['cn_share'], color=RED, marker='o', markersize=6,
          linewidth=2.2, label="China share of US imports (%)")
ax1b.set_ylabel("China's Share of US Imports (%)", color=RED, fontsize=10)
ax1b.tick_params(axis='y', labelcolor=RED)
ax1b.spines['right'].set_visible(True); ax1b.spines['top'].set_visible(False)
ax1b.set_ylim(8, 26)

for yr, lbl in events:
    ax1.axvline(yr, color=GOLD, linewidth=1.2, linestyle='--', alpha=0.7)
    ax1.text(yr, 520, lbl, ha='center', va='top', fontsize=8, color=GOLD, fontweight='bold')

for bar, val in zip(bars, df['reshoring_k']):
    ax1.text(bar.get_x()+bar.get_width()/2, bar.get_height()+5,
             f'{val}k', ha='center', fontsize=8, color=GRAY)

p1 = mpatches.Patch(color=GREEN, alpha=0.7, label='Reshoring jobs (thousands)')
p2 = plt.Line2D([0],[0], color=RED, marker='o', markersize=5, label="China's share of US imports (%)")
ax1.legend(handles=[p1,p2], loc='upper left', fontsize=9)
ax1.set_title('Reshoring Momentum and China Sourcing Decline (2015-2025)',
              fontweight='bold', fontsize=13, pad=10)
plt.setp(ax1.get_xticklabels(), visible=False)

ax2.fill_between(years, 100, df['nearshore_idx'], alpha=0.25, color=BLUE)
ax2.plot(years, df['nearshore_idx'], color=BLUE, linewidth=2.0, marker='s', markersize=5)
ax2.axhline(100, color=LGRAY, linewidth=0.8, linestyle=':')
ax2.set_ylabel('Nearshoring Index\n(2015=100)', fontsize=9)
ax2.set_xlabel('Year', fontsize=10); ax2.set_xticks(years)
ax2.set_title('Nearshoring Activity Index', fontsize=9, pad=4)
for yr, _ in events:
    ax2.axvline(yr, color=GOLD, linewidth=1.0, linestyle='--', alpha=0.5)
ax2.text(2024.5,198, '+90% vs 2015', fontsize=9, color=BLUE, fontweight='bold', ha='center')

fig.text(0.5, -0.04,
         'Source: Reshoring Initiative (2025); U.S. Census Bureau; Kearney Reshoring Index (2025)',
         ha='center', fontsize=8, color=LGRAY)
