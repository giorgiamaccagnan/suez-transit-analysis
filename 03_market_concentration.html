Chart 3 — Container Shipping Market Concentration (2025)¶

Thesis Section: 1.2 — The Physical Infrastructure of Global Trade

Data Sources:

  • Container News, "Shipping Alliances, Carriers and MSC Control Almost 83% of Market," 2025
  • Alcott Global, "Ranking the Global Shipping Giants: Capacity 2024–2025," 2025

Description: Horizontal bar chart showing the market share of the four major shipping alliances/groups plus independent carriers. Highlights the extreme concentration of the container shipping industry (83% controlled by 4 groups).

import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches
import seaborn as sns

sns.set_theme(style="whitegrid", font="serif")
plt.rcParams.update({
    'font.family': 'serif',
    'font.serif': ['Times New Roman', 'DejaVu Serif'],
    'font.size': 11,
    'axes.labelsize': 12,
    'axes.titlesize': 14,
    'figure.dpi': 150,
    'savefig.dpi': 300,
    'savefig.bbox': 'tight'
})

df = pd.read_csv('../data/carrier_market_share.csv')

# Clean carrier names for display
display_names = [
    'Ocean Alliance\n(CMA CGM, COSCO, Evergreen, OOCL)',
    'MSC\n(Independent)',
    'Gemini Cooperation\n(Maersk, Hapag-Lloyd)',
    'Premier Alliance\n(ONE, HMM, Yang Ming)',
    'Other Carriers'
]

fig, ax = plt.subplots(figsize=(12, 5.5))

colors = ['#1f4e79', '#c00000', '#2e75b6', '#4472c4', '#a6a6a6']
bars = ax.barh(range(len(df)), df['share_pct'], color=colors, edgecolor='white', height=0.65)

ax.set_yticks(range(len(df)))
ax.set_yticklabels(display_names, fontsize=10)
ax.invert_yaxis()
ax.set_xlabel('Market Share (%)', fontweight='bold')
ax.set_title('Container Shipping Market Concentration (2025)', fontweight='bold', pad=15)
ax.set_xlim(0, 35)

# Add percentage labels
for i, (bar, val) in enumerate(zip(bars, df['share_pct'])):
    ax.text(val + 0.5, bar.get_y() + bar.get_height()/2, f'{val}%', 
            va='center', fontweight='bold', fontsize=11, color=colors[i])

# Add concentration annotation
ax.axvline(x=0, color='black', linewidth=0.5)
ax.text(28, 4.8, 'Top 4 groups: 82.9%\nof global capacity', fontsize=10, 
        ha='center', style='italic',
        bbox=dict(boxstyle='round,pad=0.4', facecolor='#fff2cc', edgecolor='#d4a017', alpha=0.9))

plt.tight_layout()
plt.savefig('../figures/fig_1_2_market_concentration.png')
plt.savefig('../figures/fig_1_2_market_concentration.pdf')
plt.show()
print("✅ Chart saved to figures/fig_1_2_market_concentration.png and .pdf")
