Chart 5 — Suez Canal Monthly Transits: The Collapse and Partial Recovery¶

Thesis Sections: 1.3 / 1.4 — Chokepoints & Convergent Disruptions

Data Sources:

  • UNCTAD, Review of Maritime Transport 2024, Chapter 2
  • UNCTAD, "Navigating Troubled Waters," Rapid Assessment, February 2024
  • IMF PortWatch, Suez Canal daily transit data, 2023–2026
  • Author's compilation from Leth Agencies Suez Canal reports

Description: Time-series chart showing monthly vessel transits through the Suez Canal from January 2023 to May 2026. Colour-coded by phase: pre-crisis, crisis onset, full crisis, partial recovery, and the Hormuz escalation shock.

import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.dates as mdates
import matplotlib.patches as mpatches
import seaborn as sns

sns.set_theme(style="whitegrid", font="serif")
plt.rcParams.update({
    'font.family': 'serif',
    'font.serif': ['Times New Roman', 'DejaVu Serif'],
    'font.size': 11,
    'axes.labelsize': 12,
    'axes.titlesize': 14,
    'figure.dpi': 150,
    'savefig.dpi': 300,
    'savefig.bbox': 'tight'
})

df = pd.read_csv('../data/suez_transits_monthly.csv')
df['date'] = pd.to_datetime(df['month'])

fig, ax = plt.subplots(figsize=(14, 6))

# Color mapping
phase_colors = {
    'Pre-crisis': '#2e75b6',
    'Crisis onset': '#e67e22',
    'Crisis': '#c00000',
    'Partial recovery': '#27ae60',
    'Hormuz escalation': '#8e44ad'
}

# Plot line
ax.plot(df['date'], df['transits'], color='#333333', linewidth=1.5, zorder=2)

# Scatter with phase colors
for phase, color in phase_colors.items():
    mask = df['year_period'] == phase
    ax.scatter(df.loc[mask, 'date'], df.loc[mask, 'transits'], 
              color=color, s=50, zorder=3, edgecolors='white', linewidth=0.5)

# Shaded regions
import matplotlib.dates as mdates
crisis_start = pd.Timestamp('2023-11-01')
crisis_end = pd.Timestamp('2025-01-31')
recovery_end = pd.Timestamp('2026-02-28')

ax.axvspan(crisis_start, crisis_end, alpha=0.08, color='red')
ax.axvspan(pd.Timestamp('2026-02-01'), pd.Timestamp('2026-05-31'), alpha=0.08, color='purple')

# Annotations
ax.annotate('Houthi attacks begin\n(Nov 2023)', xy=(pd.Timestamp('2023-11-15'), 1950), 
           xytext=(pd.Timestamp('2023-06-01'), 2400),
           fontsize=9, arrowprops=dict(arrowstyle='->', color='#c00000', lw=1.5),
           bbox=dict(boxstyle='round,pad=0.3', facecolor='#fce4e4', edgecolor='#c00000', alpha=0.85))

ax.annotate('Trough: ~680/month\n(−69% from peak)', xy=(pd.Timestamp('2024-12-01'), 680), 
           xytext=(pd.Timestamp('2025-04-01'), 400),
           fontsize=9, arrowprops=dict(arrowstyle='->', color='#c00000', lw=1.5),
           bbox=dict(boxstyle='round,pad=0.3', facecolor='#fce4e4', edgecolor='#c00000', alpha=0.85))

ax.annotate('Hormuz escalation\n(Mar 2026)', xy=(pd.Timestamp('2026-03-01'), 850), 
           xytext=(pd.Timestamp('2025-10-01'), 600),
           fontsize=9, arrowprops=dict(arrowstyle='->', color='#8e44ad', lw=1.5),
           bbox=dict(boxstyle='round,pad=0.3', facecolor='#f0e0f5', edgecolor='#8e44ad', alpha=0.85))

# Pre-crisis average line
pre_avg = df[df['year_period'] == 'Pre-crisis']['transits'].mean()
ax.axhline(y=pre_avg, color='#2e75b6', linestyle='--', alpha=0.5, linewidth=1)
ax.text(pd.Timestamp('2023-01-15'), pre_avg + 50, f'Pre-crisis avg: {pre_avg:.0f}', 
       fontsize=9, color='#2e75b6')

ax.set_xlabel('Month')
ax.set_ylabel('Monthly Vessel Transits', fontweight='bold')
ax.set_title('Suez Canal Monthly Transits: Collapse and Partial Recovery (2023–2026)', 
            fontweight='bold', pad=15)

# Legend
legend_handles = [mpatches.Patch(color=c, label=l) for l, c in phase_colors.items()]
ax.legend(handles=legend_handles, loc='upper right', framealpha=0.9, fontsize=9)

ax.xaxis.set_major_formatter(mdates.DateFormatter('%b %Y'))
ax.xaxis.set_major_locator(mdates.MonthLocator(interval=3))
plt.xticks(rotation=45)
ax.set_ylim(300, 2500)

plt.tight_layout()
plt.savefig('../figures/fig_1_4_suez_transits.png')
plt.savefig('../figures/fig_1_4_suez_transits.pdf')
plt.show()
print("✅ Chart saved to figures/fig_1_4_suez_transits.png and .pdf")
