fig = plt.figure(figsize=(14.5, 7.6))
gs = GridSpec(
    1, 2,
    figure=fig,
    wspace=0.42
)
ax1 = fig.add_subplot(gs[0])
ax2 = fig.add_subplot(gs[1])
y = np.arange(len(categories))
bar_h = 0.22
for i in range(len(categories)):
    ax1.barh(
        i + bar_h,
        air_2024[i],
        height=bar_h * 0.9,
        color=RED,
        alpha=0.45,
        edgecolor='white'
    )
    ax1.barh(
        i + bar_h,
        sea_2024[i],
        height=bar_h * 0.9,
        left=air_2024[i],
        color=GREEN,
        alpha=0.4,
        edgecolor='white'
    )
    ax1.barh(
        i,
        air_2025[i],
        height=bar_h * 0.9,
        color=RED,
        alpha=0.85,
        edgecolor='white'
    )

    ax1.barh(
        i,
        sea_2025[i],
        height=bar_h * 0.9,
        left=air_2025[i],
        color=GREEN,
        alpha=0.85,
        edgecolor='white'
    )

target_specs = {
    0: {
        'arrow_y': -0.32,
        'text_x': 42,
        'text_y': -0.48,
        'ha': 'right'
    },
    1: {
        'arrow_y': 0.70,
        'text_x': 20,
        'text_y': 0.63,
        'ha': 'left'
    },
    2: {
        'arrow_y': 1.62,
        'text_x': 42,
        'text_y': 1.53,
        'ha': 'right'
    }
}
for i in range(len(categories)):
    target_air_share = 100 - target_2030[i]
    current_sea_share = sea_2025[i]

    spec = target_specs[i]
    ax1.annotate(
        '',
        xy=(target_air_share, spec['arrow_y']),
        xytext=(current_sea_share, spec['arrow_y']),
        arrowprops=dict(
            arrowstyle='->',
            color=BLUE,
            lw=1.5
        )
    )
    ax1.text(
        spec['text_x'],
        spec['text_y'],
        f'Target: {target_2030[i]}% sea',
        fontsize=8,
        color=BLUE,
        ha=spec['ha'],
        va='top',
        fontweight='bold'
    )
ax1.axvline(
    50,
    color=LGRAY,
    linewidth=0.7,
    linestyle=':'
)
ax1.set_yticks(y)
ax1.set_yticklabels(categories, fontsize=10)
