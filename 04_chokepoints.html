import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches
import numpy as np
import seaborn as sns

sns.set_theme(style="whitegrid", font="serif")
plt.rcParams.update({
    'font.family': 'serif',
    'font.serif': ['Times New Roman', 'DejaVu Serif'],
    'font.size': 11,
    'axes.labelsize': 12,
    'axes.titlesize': 14,
    'figure.dpi': 150,
    'savefig.dpi': 300,
    'savefig.bbox': 'tight'
})

df = pd.read_csv('../data/chokepoints.csv')


df = df.sort_values('share_global_trade_pct', ascending=True)

fig, ax = plt.subplots(figsize=(12, 6))

colors = {
    'Strait of Malacca': '#1f4e79',
    'Strait of Hormuz': '#c00000',
    'Suez Canal': '#e67e22',
    'Bab al-Mandab': '#d4a017',
    'Panama Canal': '#2e75b6'
}

bar_colors = [colors[name] for name in df['chokepoint']]
bars = ax.barh(df['chokepoint'], df['share_global_trade_pct'], color=bar_colors, 
               edgecolor='white', height=0.6)


details = {
    'Strait of Malacca': '94,301 transits · 45% oil · 2.8 km wide',
    'Strait of Hormuz': '20.9M bbl/day · 39% oil · 54 km wide',
    'Suez Canal': '~26,000 transits/yr · 22% containers · 0.2 km wide',
    'Bab al-Mandab': '20,000+ transits/yr · 9.3% oil · 26 km wide',
    'Panama Canal': '~10,200 transits · drought vulnerable · 0.15 km wide'
}

for bar, name in zip(bars, df['chokepoint']):
    width = bar.get_width()
    ax.text(width + 0.5, bar.get_y() + bar.get_height()/2 - 0.08, 
            f'{width}%', va='center', fontweight='bold', fontsize=12)
    ax.text(width + 0.5, bar.get_y() + bar.get_height()/2 + 0.22,
            details[name], va='center', fontsize=8.5, color='#555555', style='italic')

ax.set_xlabel('Share of Global Seaborne Trade (%)', fontweight='bold')
ax.set_title("The World's Five Critical Maritime Chokepoints", fontweight='bold', pad=15)
ax.set_xlim(0, 35)


ax.text(25, 0.5, 'Any single disruption\nsends global shockwaves', fontsize=10,
        ha='center', va='center',
        bbox=dict(boxstyle='round,pad=0.5', facecolor='#fce4e4', edgecolor='#c00000', alpha=0.8))

plt.tight_layout()
plt.savefig('../figures/fig_1_3_chokepoints.png')
plt.savefig('../figures/fig_1_3_chokepoints.pdf')
plt.show()
print("✅ Chart saved to figures/fig_1_3_chokepoints.png and .pdf")
