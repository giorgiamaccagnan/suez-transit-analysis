fig, ax = plt.subplots(figsize=(15, 7.5))

phases = [
    ('2023-10-15','2023-12-15','#fff3e0','Phase 1: Initial shock'),
    ('2023-12-15','2024-07-15','#fce4ec','Phase 2: Escalation'),
    ('2024-07-15','2025-12-15','#e8f5e9','Phase 3: Normalization'),
    ('2025-12-15','2026-05-15','#fce4ec','Phase 4: Hormuz shock'),
]
for s, e, col, lbl in phases:
    start, end = pd.Timestamp(s), pd.Timestamp(e)
    ax.axvspan(start, end, alpha=0.25, color=col, zorder=0)
    
    ax.text(start + (end - start) / 2, 8200, lbl,
            ha='center',
            va='center',
            fontsize=8,
            color=GRAY,
            style='italic')

ax.axhline(8700, color=GOLD, linewidth=1.2, linestyle=':', alpha=0.7)
ax.text(pd.Timestamp("2026-05-01"), 8880, 'COVID-19 peak ≈ $8,700',
        fontsize=8, color=GOLD, ha='right', fontweight='bold')
ax.axhline(707,  color=BLUE, linewidth=1.0, linestyle=':', alpha=0.5)
ax.text(df['date'].iloc[0], 400, 'Pre-crisis baseline $707',
        fontsize=8, color=BLUE)

ax.fill_between(df['date'], 0, df['scfi'], alpha=0.12, color=RED)
ax.plot(df['date'], df['scfi'], color=RED, linewidth=2.5, zorder=4)
ax.scatter(df['date'], df['scfi'], color='white', s=55,
           zorder=5, edgecolors=RED, linewidth=2.0)

for date_str, val, label, text_date_str, text_y, align in events:

    ax.annotate(

        xy=(d, val),
        xytext=(text_d, text_y),
