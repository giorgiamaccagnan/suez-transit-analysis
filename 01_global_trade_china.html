Chart 1 — Global Trade Explosion and China's Rise (1990–2023)¶

Thesis Section: 1.1 — Globalisation and the Lean Supply Chain

Data Sources:

  • World Bank, World Development Indicators: NE.EXP.GNFS.CD (Exports of goods and services, current USD)
  • World Bank, World Development Indicators: NE.EXP.GNFS.CD for China
  • Author's calculations for China's share

Description: Dual-axis chart showing the growth of world merchandise exports (left axis, USD trillions) alongside China's rising share of global exports (right axis, %). Highlights China's WTO accession in 2001 and the 2008 financial crisis.

import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.ticker as mticker
import seaborn as sns

# Style
sns.set_theme(style="whitegrid", font="serif")
plt.rcParams.update({
    'font.family': 'serif',
    'font.serif': ['Times New Roman', 'DejaVu Serif'],
    'font.size': 11,
    'axes.labelsize': 12,
    'axes.titlesize': 14,
    'figure.dpi': 150,
    'savefig.dpi': 300,
    'savefig.bbox': 'tight'
})

# Load data
df = pd.read_csv('../data/world_china_exports.csv')

fig, ax1 = plt.subplots(figsize=(12, 6))

# Left axis: World exports (trillions)
color_world = '#1f4e79'
ax1.fill_between(df['year'], df['world_exports_usd_trillion'], alpha=0.15, color=color_world)
ax1.plot(df['year'], df['world_exports_usd_trillion'], color=color_world, linewidth=2.2, 
         label='World Exports (USD trillions)', marker='o', markersize=3)
ax1.set_xlabel('Year')
ax1.set_ylabel('World Exports (USD trillions)', color=color_world, fontweight='bold')
ax1.tick_params(axis='y', labelcolor=color_world)
ax1.set_ylim(0, 32)

# Right axis: China's share
ax2 = ax1.twinx()
color_china = '#c00000'
ax2.plot(df['year'], df['china_share_pct'], color=color_china, linewidth=2.2, linestyle='--',
         label="China's Share of World Exports (%)", marker='s', markersize=3)
ax2.set_ylabel("China's Share (%)", color=color_china, fontweight='bold')
ax2.tick_params(axis='y', labelcolor=color_china)
ax2.set_ylim(0, 20)

# Put ax1 annotations above ax2
ax1.set_zorder(ax2.get_zorder() + 1)
ax1.patch.set_visible(False)

# Annotations
ax1.axvline(x=2001, color='gray', linestyle=':', alpha=0.7)
ax1.annotate('China joins WTO\n(Dec 2001)', xy=(2001, 7.5), xytext=(2003, 10),
            fontsize=9, ha='left', arrowprops=dict(arrowstyle='->', color='gray'),
            bbox=dict(boxstyle='round,pad=0.3', facecolor='lightyellow', edgecolor='gray', alpha=0.8))

ax1.axvline(x=2008, color='gray', linestyle=':', alpha=0.7, zorder=5)

ax1.annotate(
    'Global Financial\nCrisis (2008)',
    xy=(2008, 20),
    xytext=(2005, 23),
    fontsize=9,
    ha='center',
    zorder=20,
    arrowprops=dict(arrowstyle='->', color='gray', zorder=20),
    bbox=dict(
        boxstyle='round,pad=0.3',
        facecolor='lightyellow',
        edgecolor='gray',
        alpha=0.95,
        zorder=20
    )
)

ax1.axvline(x=2020, color='gray', linestyle=':', alpha=0.7)
ax1.annotate('COVID-19', xy=(2020, 22), xytext=(2018, 26),
            fontsize=9, ha='center', arrowprops=dict(arrowstyle='->', color='gray'),
            bbox=dict(boxstyle='round,pad=0.3', facecolor='lightyellow', edgecolor='gray', alpha=0.8))

# Combined legend
lines1, labels1 = ax1.get_legend_handles_labels()
lines2, labels2 = ax2.get_legend_handles_labels()
ax1.legend(lines1 + lines2, labels1 + labels2, loc='upper left', framealpha=0.9, fontsize=10)

ax1.set_title('Global Trade Explosion and China\'s Rise (1990–2023)', fontweight='bold', pad=15)
ax1.set_xlim(1990, 2023)

plt.tight_layout()
plt.savefig('../figures/fig_1_1_global_trade_china.png')
plt.savefig('../figures/fig_1_1_global_trade_china.pdf')
plt.show()
print("✅ Chart saved to figures/fig_1_1_global_trade_china.png and .pdf")
